#include <stdio.h>//incluindo biblioteca stdio.h (necessária para a execução do programa).

typedef struct contaBancaria {//declaração uma estrutura do tipo conta bancaria.
  char nome [100];//declaração de variavel "nome" do tipo char.
  int numero;//declaração de variavel "numero" do tipo int.
  float saldo;//declaração de variavel "saldo" do tipo float.
}contaBancaria;// renomeando a estrutura para o nome "contaBancaria".

void deposita(float valor, struct contaBancaria* conta){//declaração de função com 2 parametros que deposita um novo valor ao saldo.
conta->saldo = conta->saldo + valor;//atribuição de um novo valor para a variavel saldo.
}

int main(void) {//implementando a função principal.
float valor = 10;//adicionando um valor para a variavel de nome "valor".
contaBancaria a;//declaração de uma variavel chamada "a" do tipo contaBancaria.
printf("Digite seu nome: ");//pede ao usuario que digite seu nome.
scanf("%[^\n]", a.nome);//recebe o nome do usuario e armazena na variavel "nome" da struct contaBancaria.
printf("Digite seu numero: ");//pede ao usuario que digite seu numero.
scanf("%d", &a.numero);//recebe o numero do usuario e armazena na variavel "numero" da struct contaBancaria.
printf("Digite seu saldo: ");//pede ao usuario que digite seu saldo.
scanf("%f", &a.saldo);//recebe o saldo do usuario e armazena na variavel "saldo" da struct contaBancaria.

deposita (valor, &a);//chamada da função deposita.
printf("Novo saldo: %2.f\n", a.saldo);//imprime o novo saldo do usuario.

  
  return 0;
}
